Cognit vs JavaScript

Cognit
fn fizzbuzz(n) {
    if n % 15 == 0 -> "FizzBuzz"
    else if n % 3 == 0 -> "Fizz"
    else if n % 5 == 0 -> "Buzz"
    else -> n
}

for i in range(1, 16) {
    print(fizzbuzz(i))
}
JavaScript
function fizzbuzz(n) {
    if (n % 15 === 0) return "FizzBuzz"
    if (n % 3 === 0) return "Fizz"
    if (n % 5 === 0) return "Buzz"
    return n
}

for (const i of range(1, 16)) {
    console.log(fizzbuzz(i))
}

Pattern Matching

Cognitfn describe(n) {
    match n {
        0 -> "zero"
        1 -> "one"
        2 -> "two"
        _ -> "many"
    }
}
Generated JSfunction describe(n) {
    return (function() {
        if (n === 0) { return "zero"; }
        if (n === 1) { return "one"; }
        if (n === 2) { return "two"; }
        { return "many"; }
    })();
}
